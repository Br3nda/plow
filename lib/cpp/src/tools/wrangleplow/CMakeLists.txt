

PROJECT(PLOW_WRANGLEPLOW)

INCLUDE(${QT_USE_FILE})

SET(WRANGLE_PLOW_SRCS_CXX
    wrangleplow.cpp
    main.cpp
)

SET(WRANGLE_PLOW_MOC_HDRS
    wrangleplow.h
)

SET(WRANGLE_PLOW_UI_FILES

)

QT4_WRAP_UI( WRANGLE_PLOW_SRCS_CXX ${WRANGLE_PLOW_UI_FILES} )
QT4_WRAP_CPP( WRANGLE_PLOW_SRCS_CXX ${WRANGLE_PLOW_MOC_HDRS} )

SOURCE_GROUP("Generated Files" REGULAR_EXPRESSION "(ui_.*h)$")
SOURCE_GROUP("UI Files" REGULAR_EXPRESSION ".*.ui$")

IF(APPLE)
  ADD_EXECUTABLE(wrangleplow MACOSX_BUNDLE ${WRANGLE_PLOW_SRCS_CXX})
ELSEIF(UNIX)
  ADD_EXECUTABLE(wrangleplow ${WRANGLE_PLOW_SRCS_CXX})
ENDIF()

TARGET_LINK_LIBRARIES(wrangleplow ${QT_LIBRARIES} plow plowgui)
